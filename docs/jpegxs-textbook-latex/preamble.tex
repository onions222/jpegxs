% ── Code Listings (our style, overrides ElegantBook's basic one) ──
\definecolor{codegreen}{RGB}{105,130,105}
\definecolor{codegray}{RGB}{140,140,140}
\definecolor{codepurple}{RGB}{130,110,140}
\definecolor{backcolour}{RGB}{248,247,245}
\definecolor{codeframe}{RGB}{175,172,165}

\lstdefinestyle{codestyle}{
    backgroundcolor=\color{backcolour},
    commentstyle=\color{codegreen},
    keywordstyle=\color{codepurple}\bfseries,
    numberstyle=\tiny\color{codegray},
    stringstyle=\color{codegreen},
    basicstyle=\ttfamily\footnotesize,
    breakatwhitespace=false,
    breaklines=true,
    captionpos=b,
    keepspaces=true,
    numbers=left,
    numbersep=5pt,
    showspaces=false,
    showstringspaces=false,
    showtabs=false,
    tabsize=2,
    frame=single,
    rulecolor=\color{codeframe},
    xleftmargin=2em,
    framexleftmargin=1.5em,
    escapeinside={(*@}{@*)},
}

\lstset{style=codestyle}

% C language definition

              if,else,for,while,do,switch,case,break,continue,

              assert,true,false,NULL,size_t,ptrdiff_t,
              uint8_t,uint16_t,uint32_t,uint64_t,
              int8_t,int16_t,int32_t,int64_t},

    comment=[l]{//},
    morecomment=[s]{/*}{*/},
    commentstyle=\color{codegreen},
    stringstyle=\color{codegreen},
    morestring=[b]",
}

% ── Unified TikZ Flowchart Styles ─────────────────────────────────
\tikzset{
    flowblock/.style={
        rectangle, draw=structurecolor!70, fill=structurecolor!10,
        text width=#1, align=center,
        minimum height=0.7cm, font=\small,
        inner sep=4pt
    },
    flowblock/.default=3.8cm,
    flowarrow/.style={-{Stealth[length=2.5mm]}, thick, draw=structurecolor!60},
    flowdecision/.style={
        diamond, draw=second!70, fill=second!12,
        text width=#1, align=center,
        aspect=2, font=\small, inner sep=2pt
    },
    flowdecision/.default=2cm,
}
